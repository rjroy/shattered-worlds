Implementation sketch

// src/game/scenes/WorldSelectScene.ts
import Phaser from 'phaser'
import titleBgUrl from '../assets/title.webp'   // Vite static import
import { worldManifest } from '../../data/worldManifest'
import { worldDisplayManifest } from '../../data/worldDisplayManifest'
import { selectTheme } from '../view/themes/themeManifest'
import { textStyle, CANVAS_W, CANVAS_H } from '../view/presentation'

const CARD_W = 240
const CARD_H = 350
const CARD_GAP = 30
const CARD_Y = 390        // lower half — over the stone path area of the image
const SUBTITLE_Y = 555    // below cards; adjust to taste

export class WorldSelectScene extends Phaser.Scene {
  constructor() { super({ key: 'WorldSelect' }) }

  preload(): void {
    this.load.image('world-select-bg', titleBgUrl)
  }

  create(): void {
    // title image — scaled to fill the canvas
    this.add.image(CANVAS_W / 2, CANVAS_H / 2, 'world-select-bg')
      .setDisplaySize(CANVAS_W, CANVAS_H)

    // subtitle only — title logotype is in the image
    this.add.text(CANVAS_W / 2, SUBTITLE_Y, 'Choose your shard',
      textStyle({ fontSize: '13px', color: '#9aa3b2' })
    ).setOrigin(0.5, 0.5)

    // world cards
    const worldIds = Object.keys(worldManifest)
    const totalW = worldIds.length * CARD_W + (worldIds.length - 1) * CARD_GAP
    const startX = (CANVAS_W - totalW) / 2 + CARD_W / 2

    worldIds.forEach((worldId, i) => {
      const display = worldDisplayManifest[worldId]
      if (display === undefined) {
        throw new Error(`WorldSelectScene: no display entry for worldId "${worldId}"`)
      }
      const theme = selectTheme(worldId)
      const accentColor = Phaser.Display.Color.HexStringToColor(theme.intrusionHue).color
      const cardX = startX + i * (CARD_W + CARD_GAP)

      this.createWorldCard(worldId, cardX, CARD_Y, display.name, display.tagline, display.story, accentColor)
    })
  }

  private createWorldCard(
    worldId: string,
    cx: number, cy: number,
    name: string, tagline: string, story: string,
    accentColor: number,
  ): void {
    const container = this.add.container(cx, cy)

    // background + accent border
    const bg = this.add.rectangle(0, 0, CARD_W, CARD_H, 0x181c28)
    const border = this.add.rectangle(0, 0, CARD_W, CARD_H)
    border.setStrokeStyle(2, accentColor)
    border.setFillStyle()  // transparent fill

    // content: name, tagline, story
    const nameText = this.add.text(0, -CARD_H / 2 + 30, name,
      textStyle({ fontSize: '17px', color: '#c8cde0', fontStyle: 'bold' })
    ).setOrigin(0.5, 0)

    const tagText = this.add.text(0, -CARD_H / 2 + 60, tagline,
      textStyle({ fontSize: '12px', color: '#6b7099', fontStyle: 'italic' })
    ).setOrigin(0.5, 0)

    const storyText = this.add.text(0, -CARD_H / 2 + 90, story,
      textStyle({ fontSize: '12px', color: '#a0a8c0',
        wordWrap: { width: CARD_W - 24 } })
    ).setOrigin(0.5, 0)

    container.add([bg, border, nameText, tagText, storyText])

    // interactivity on the background rect
    bg.setInteractive({ useHandCursor: true })
    bg.on('pointerover',  () => container.setScale(1.03))
    bg.on('pointerout',   () => container.setScale(1.0))
    bg.on('pointerdown',  () => {
      const seed = Math.floor(Math.random() * 2 ** 32)
      this.scene.start('Table', { worldId, seed })
    })
  }
}
Layout note: CARD_Y = 390 and SUBTITLE_Y = 555 are starting estimates designed to sit the cards over the dark stone-path area of title.webp. The text positions within the card (name at -145, tagline at -115, story at -85 from container center, derived from -CARD_H/2 + {30, 60, 90}) need browser review — story text height is dynamic. Shift CARD_Y up if the subtitle overlaps the canvas bottom, or reduce CARD_H. The title image's portal and figure occupy roughly y=130–480 at canvas center, so pushing cards to the right-and-lower avoids occlusion.
Interactivity hitbox: Pointer events are registered on bg (the background rectangle), but Text objects sit above it in the container's display list. In Phaser 3, a non-interactive Text object does not block pointer events from reaching objects beneath it — so the bg rect should receive events across the full card area. Verify this during browser testing by hovering over text regions and confirming the scale emphasis fires. If it doesn't, the fix is to make text objects explicitly non-interactive (textObj.setInteractive() is not called on them already, so they should pass through correctly).
Tween caution: Do not use scene.tweens.add for hover scale — use direct setScale instead. A prior retro (visual-identity-rollout) recorded a bug where Phaser recycles tweens from a pool, causing stale tweens to apply unexpectedly when the same object is tweened rapidly. The direct setScale approach avoids this entirely for a simple hover effect.
